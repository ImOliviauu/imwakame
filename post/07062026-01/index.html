A. 发消息

题目要求在给定总时间 NNN 的情况下，寻找一个正整数 TTT，使得 AA（奇数倍时刻在线）和 BA（偶数倍时刻在线）在发送消息时全部处于各自的在线区间内 。

通过给定的一组在线区间，我们可以求补集得到 AA 的离线区间集合 VAV_AVA​ 和 BA 的离线区间集合 VBV_BVB​ 。
考虑暴力，对于任意选定的 TTT：

  • 不能存在奇数 iii 使得 iT∈[l,r]∈VAiT \in [l,r] \in V_AiT∈[l,r]∈VA​。
  • 不能存在偶数 iii 使得 iT∈[l,r]∈VBiT \in [l,r] \in V_BiT∈[l,r]∈VB​。

由于 N≤109N \le 10^9N≤109 ，直接枚举 TTT 及其所有倍数显然会超时。注意到倍数 iii 和基础时间 TTT 的乘积不超过 NNN，考虑根号分治 。

设定阈值 B=31700B=31700B=31700 。

对于 T≤BT\le BT≤B 我们直接如上枚举，容易做到 O(1)O(1)O(1) 检查。

当 T>BT > BT>B 时，发消息的系数 iii 必定满足 i≤⌊NB⌋≤Bi \le \lfloor \frac{N}{B} \rfloor \le Bi≤⌊BN​⌋≤B ，所以枚举系数 iii。

  • 对于 VAV_AVA​ 的每个区间 [l,r][l,r][l,r]，枚举奇数 i∈[1,B]i \in [1,B]i∈[1,B] 。
  • 对于 VBV_BVB​ 的每个区间 [l,r][l,r][l,r]，枚举偶数 i∈[2,B]i \in [2,B]i∈[2,B] 。

对于固定的区间 [l,r][l,r][l,r] 和系数 iii，导致非法的 TTT 满足 l≤i⋅T≤rl \le i \cdot T \le rl≤i⋅T≤r，即：

⌈li⌉≤T≤⌊ri⌋\lceil \frac{l}{i} \rceil \le T \le \lfloor \frac{r}{i} \rfloor⌈il​⌉≤T≤⌊ir​⌋

⌊l+i−1i⌋≤T≤⌊ri⌋\lfloor \frac{l+i-1}{i} \rfloor \le T \le \lfloor \frac{r}{i} \rfloor⌊il+i−1​⌋≤T≤⌊ir​⌋

我们将产生的非法区间 [le,ri][le,ri][le,ri] 限制在 T>BT > BT>B 的范围内，即 le=max⁡(le,B+1)le = \max(le,B+1)le=max(le,B+1) 。然后给区间去一下重计数即可

#include<bits/stdc++.h>

using namespace std;

#define ll long long

const int N=305;
const int B=31700;

int n,X,Y;
array<int,2> a[N],b[N];
vector<array<int,2> > va,vb;
vector<array<int,2> > no,no1;
int ans;

bool chk(int x)
{
    for(auto i:va)
    {
        int l=i[0],r=i[1];
        if((l+x-1)/(2*x)!=(r+x)/(2*x)) return 0;
    }
    for(auto i:vb)
    {
        int l=i[0],r=i[1];
        if((l-1)/(2*x)!=r/(2*x)) return 0;
    }
    return 1;
}

B. 旅行

给定一棵有根树，从节点 uuu 出发走向其子树中的 vvv。路径上的每个节点有概率被选中，若选中节点的权值 vkv_kvk​ 的最大公约数为 xxx，则获得收益 cxc_xcx​ 。求每个起点向其子树内所有点行走的期望收益之和 。

考虑对 cxc_xcx​ 容斥时的情景，与其因数有关，设 cx=∑d∣xfdc_x = \sum_{d|x} f_dcx​=∑d∣x​fd​，通过反演可得 fd=∑k∣dck⋅μ(dk)f_d = \sum_{k|d} c_k \cdot \mu(\frac{d}{k})fd​=∑k∣d​ck​⋅μ(kd​)。求 GCD 为 xxx 的期望可以转化为：求选中集合的公约数是 ddd 的倍数的概率，再乘上 fdf_dfd​ 。
对于一条从 uuu 到 vvv 的路径，选中节点均为 ddd 的倍数，且至少选中一个节点的概率为：

∏k∈path(u,v)(1−Pk⋅[d∤vk])−∏k∈path(u,v)(1−Pk)\prod_{k \in \text{path}(u,v)} (1 - P_k \cdot [d \nmid v_k]) - \prod_{k \in \text{path}(u,v)} (1 - P_k)∏k∈path(u,v)​(1−Pk​⋅[d∤vk​])−∏k∈path(u,v)​(1−Pk​)

其中 Qk=1−PkQ_k=1-P_kQk​=1−Pk​。将左侧提取出路径全集的 QkQ_kQk​ 乘积，只有当 d∣vkd \mid v_kd∣vk​ 时，项从 QkQ_kQk​ 变为 1。因此上式等于：

(∏k∈path(u,v)Qk)×(∏k∈path(u,v),d∣vkQk−1−1)\left( \prod_{k \in \text{path}(u,v)} Q_k \right) \times \left( \prod_{k \in \text{path}(u,v),d \mid v_k} Q_k^{-1} - 1 \right)(∏k∈path(u,v)​Qk​)×(∏k∈path(u,v),d∣vk​​Qk−1​−1)

令 pqxpq_xpqx​ 表示从根节点到 xxx 的 QkQ_kQk​ 连乘积 。则路径 u→vu \to vu→v 的连乘积可以写为 pqvpqparu\frac{pq_v}{pq_{par_u}}pqparu​​pqv​​ 。提取出与 vvv 相关的部分，我们需要在子树内对每个因数 ddd 求和：

∑v∈subtree(u)pqv×(∏k∈path(u,v),d∣vkQk−1−1)\sum_{v \in \text{subtree}(u)} pq_v \times \left( \prod_{k \in \text{path}(u,v),d \mid v_k} Q_k^{-1} - 1 \right)∑v∈subtree(u)​pqv​×(∏k∈path(u,v),d∣vk​​Qk−1​−1)

具体地，使用了一个全局数组 cnt[d] 来维护因子为 ddd 的路径加权贡献。在遍历树的过程中，进入节点 xxx 时，对于所有的因数 d∣vxd \mid v_xd∣vx​，先记录进入子树前的 cnt[d] 并减去 ：dp[x][i] = add(dp[x][i],mod - cnt[fac[v[x]][i]])
随后递归遍历子树，子树累加了 ∑pqy\sum pq_y∑pqy​。此时，子树内所有的路径端点都被纳入计算。我们将当前的子树和加上之前的 cnt 差分值，乘上当前节点自身的权重变化量 PxQx\frac{P_x}{Q_x}Qx​Px​​ ，并将其更新回 cnt[d] 中 。这个过程本质上是在路径上第一次遇到 ddd 的倍数时，将该点及其后继路径挂载到 ddd 的贡献池中。最后通过另一次简单的 DFS dfs1，将子树的答案累加，并乘上反演系数 cf 得到最终答案 。

ll add(ll x,ll y)
{
    x+=y;
    if(x>=mod) x-=mod;
    return x;
}

ll qpow(ll x,ll y=mod-2)
{
    ll ret=1;
    while(y)
    {
        if(y&1)
        {
            ret=ret*x%mod;
        }
        x=x*x%mod;
        y>>=1;
    }
    return ret;
}

ll gtmu(ll x)
{
    ll ret=1;
    for(ll i=2;i*i<=x;i++)
    {
        if(x%i==0)
        {
            ll c=0;
            while(x%i==0) x/=i,c++;
            if(c==1) ret=mod-ret;
            else return 0;
        }
    }
    if(x>1) ret=mod-ret;
    return ret;
}


void dfs0(ll x)
{
    pq[x]=pq[par[x]]*q[x]%mod;
    for(auto y:e[x])
    {
        dfs0(y);
    }
}

void dfs(ll x)
{
    ll m=fac[v[x]].size();
    dp[x].resize(m);
    for(ll i=0;i<m;i++)
    {
        dp[x][i]=add(dp[x][i],mod-cnt[fac[v[x]][i]]);
    }
    s[x]=pq[x];
    for(auto y:e[x])
    {
        dfs(y);
        s[x]=add(s[x],s[y]);
    }
    for(ll i=0;i<m;i++)
    {
        dp[x][i]=add(dp[x][i],cnt[fac[v[x]][i]]);
        dp[x][i]=add(dp[x][i],s[x]);
        dp[x][i]=dp[x][i]*iq[x]%mod*p[x]%mod;
        cnt[fac[v[x]][i]]=add(cnt[fac[v[x]][i]],dp[x][i]);
    }
}

void dfs1(ll x)
{
    for(auto y:e[x])
    {
        dfs1(y);
        ans[x]=add(ans[x],ans[y]);
    }
    for(ll i=0;i<fac[v[x]].size();i++)
    {
        ans[x]=add(ans[x],dp[x][i]*cf[fac[v[x]][i]]%mod);
    }
}

C.冒泡排序二合一

题目给出一个排列，每次询问给定区间 [l,r][l,r][l,r] 进行 kkk 轮冒泡排序后的结果，询问分为两种：

  1. 原本值为 xxx 的元素，排序后的下标位置 。

  2. 排序后下标为 xxx 的位置，其元素的值 。
    每次询问独立，不修改原数组 。

冒泡排序的本质物理过程是：每一轮中，每个元素会向左移动一步，除非它的左边没有比它大的元素；而作为当前遇到的较大值，它会像气泡一样向右滚动。
具体而言，进行了 kkk 轮冒泡排序后：

  • 一个元素 vvv 初始位置为 ppp，其左侧如果有 ccc 个大于 vvv 的元素，它向左移动的步数为 min⁡(k,c)\min(k,c)min(k,c)。
  • 区间最右侧的区域（具体视最大元素的数量而定）将被排序且填满区间内的最大元素。

求值 xxx 的新位置

我们在建立主席树时，按数值从大到小向对应位置插入权值 。这样 rt[x] 维护的就是所有值 ≥x\ge x≥x 的元素的位置分布。
对于询问，设 xxx 的初始位置为 pxp_xpx​（代码中的 p[x] ）。
我们在 rt[x] 版本的主席树中，查询区间 [l,n][l,n][l,n] 内，第 k+1k+1k+1 个比 xxx 大（或等于 xxx）的元素的位置 pos = T.kbig(rt[x],1,n,l,k+1) 。

若 pos<pxpos < p_xpos<px​，说明在 xxx 的左侧，至少有 k+1k+1k+1 个比它大的元素。因为大元素非常多，在 kkk 轮排序中，xxx 每轮都会稳稳地向左移动 1 步。最终位置为 px−kp_x - kpx​−k 。

若 pos>rpos > rpos>r ，说明整个区间 [l,r][l,r][l,r] 内，比 xxx 大的元素不足 k+1k+1k+1 个。此时 xxx 已经属于区间内最大的那一批数字，被挤到了区间的最右侧有序部分。具体位置即为 rrr 减去区间内严格大于 xxx 的元素个数，代码中表达为 r + 1 - T.query(rt[x],1,n,l,r)。

否则，xxx 的左侧比它大的元素不足 kkk 个，但区间内总量足够：xxx 在向左移动的过程中，其左边的大元素不够阻挡它，它最终会卡在第 k+1k+1k+1 个大元素本应所在相对位置之前。新位置即为 pos−kpos - kpos−k 。

求新下标 xxx 上的值

这次按照原位置下标从左到右建立主席树，rt[i] 维护的是前 iii 个位置的数值分布 。

若 x+k>rx + k > rx+k>r ：这个位置处于区间右侧。因为每轮都会把当前的一个最大值送到右边，kkk 轮后，右侧的这一段区域已经完全处于升序排列的状态，且存放的正是整个区间内的最大值们。这个位置存放的元素就是区间 [l,r][l,r][l,r] 内从大到小第 r−x+1r - x + 1r−x+1 大的数字。通过主席树区间求第 KKK 大 T.kth(rt[l-1],rt[r],r-x+1) 即可得到 。

若 x+k≤rx + k \le rx+k≤r ：这个位置还未达到绝对有序的状态。根据冒泡左移性质，最终能落在位置 xxx 的数，原本的下标必然在 [l,x+k][l,x+k][l,x+k] 这个区间内（因为它最多左移 kkk 步到达 xxx）。这个数要么是原本就在位置 x+kx+kx+k 的元素 a[x+k]a[x+k]a[x+k]，要么是前缀区间 [l,x+k][l,x+k][l,x+k] 内第 k+1k+1k+1 大的元素。两者取最小值即可，即 min(a[x+k],T.kth(rt[l-1],rt[x+k],k+1)) 。

#include<bits/stdc++.h>

using namespace std;

#define ll long long

const int N=6e5+1;

int n,q,o;
int a[N],p[N];
int rt[N];

struct hjttree
{
    struct node
    {
        int cnt,lch,rch;
    } tr[N*40];
    int tot;

    inline void insert(int &p,int op,int l,int r,int x)
    {
        p=++tot;
        tr[p]=tr[op];
        tr[p].cnt++;
        if(l==r) return ;
        int mid=l+r>>1;
        if(x<=mid) insert(tr[p].lch,tr[op].lch,l,mid,x);
        else insert(tr[p].rch,tr[op].rch,mid+1,r,x);
    }

    inline int query(int p,int l,int r,int x,int y)
    {
        if(x<=l&&y>=r) return tr[p].cnt;
        int mid=l+r>>1;
        if(y<=mid) return query(tr[p].lch,l,mid,x,y);
        else if(x>mid) return query(tr[p].rch,mid+1,r,x,y);
        return query(tr[p].lch,l,mid,x,y)+query(tr[p].rch,mid+1,r,x,y);
    }

    inline int kbig(int p,int l,int r,int x,int &k)
    {
        if(x<=l&&tr[p].cnt<k)
        {
            k-=tr[p].cnt;
            return n+1;
        }
        if(l==r) return l;
        int mid=l+r>>1,ret=n+1;
        if(x<=mid) ret=kbig(tr[p].lch,l,mid,x,k);
        if(ret<=n) return ret;
        return kbig(tr[p].rch,mid+1,r,x,k);
    }

    inline int kth(int l,int r,int k)
    {   
        int L=1,R=n,ret=1;
        while(L<R)
        {
            int mid=L+R>>1;
            int c=tr[tr[r].rch].cnt-tr[tr[l].rch].cnt;
            if(k>c)
            {
                k-=c;
                R=mid;
                l=tr[l].lch,r=tr[r].lch;
            }
            else
            {
                L=mid+1;
                ret=mid+1;
                l=tr[l].rch,r=tr[r].rch;
            }
        }
        return ret;
    }
} T;

                cout<<r+1-T.query(rt[x],1,n,l,r)<<"\n";
            }
            else
            {
                cout<<pos-k<<"\n";
            }
        }
    }
    else
    {
        for(int i=1;i<=n;i++)
        {
            T.insert(rt[i],rt[i-1],1,n,a[i]);
        }
        while(q--)
        {
            int l,r,k,x;
            cin>>l>>r>>k>>x;
            if(x<l||x>r)
            {
                cout<<a[x]<<"\n";
                continue;
            }
            if(x+k>r)
            {
                cout<<T.kth(rt[l-1],rt[r],r-x+1)<<"\n";
            }
            else
            {
                cout<<min(a[x+k],T.kth(rt[l-1],rt[x+k],k+1))<<"\n";
            }
        }
    }

    return 0;
}
